Key Differences

  • Number Methods: Primarily focus on numeric representation and formatting
  • String Methods: Focus on text manipulation, searching, and extraction
  • Numbers are primitive values but have object wrappers with useful methods
  • Strings are immutable - methods return new strings rather than modifying the original
  • Many string methods accept regular expressions for powerful pattern matching

Number Methods

// toString() - converts number to string
let num = 15;
console.log(num.toString()); // "15"
console.log(typeof num.toString()); // "string"
console.log((10).toString(2)); // "1010" (binary representation)

// toFixed() - formats number with fixed decimal places
let pi = 3.14159;
console.log(pi.toFixed(2)); // "3.14"
console.log(pi.toFixed(0)); // "3"
console.log(2.55.toFixed(1)); // "2.6" (rounds up)

// toExponential() - returns string in exponential notation
console.log(123456.toExponential()); // "1.23456e+5"
console.log(123456.toExponential(2)); // "1.23e+5"

// toPrecision() - formats to specified precision
let num2 = 123.456;
console.log(num2.toPrecision(5)); // "123.46"
console.log(num2.toPrecision(2)); // "1.2e+2"

// Number() - converts other types to numbers
console.log(Number("123")); // 123
console.log(Number("123.45")); // 123.45
console.log(Number(true)); // 1
console.log(Number("hello")); // NaN

// parseInt() and parseFloat() - string parsing
console.log(parseInt("123px")); // 123
console.log(parseFloat("12.34em")); // 12.34
console.log(parseInt("101", 2)); // 5 (binary parsing)

// Number properties
console.log(Number.MAX_SAFE_INTEGER); // 9007199254740991
console.log(Number.MIN_SAFE_INTEGER); // -9007199254740991
console.log(Number.POSITIVE_INFINITY); // Infinity

Characteristics

  • Number methods don't change the original number (numbers are immutable)
  • Most methods return string representations of the number
  • Be cautious with floating point precision issues in JavaScript
  • Use parseInt() with radix parameter for different number bases
  • Check for NaN with isNaN() or Number.isNaN()

String Methods

// charAt() - returns character at specified index
let str = "JavaScript";
console.log(str.charAt(0)); // "J"
console.log(str.charAt(4)); // "S"

// concat() - joins strings
console.log("Hello".concat(" ", "World")); // "Hello World"

// includes() - checks if string contains substring
console.log("JavaScript".includes("Script")); // true
console.log("JavaScript".includes("script")); // false (case-sensitive)

// indexOf() and lastIndexOf() - find substring positions
console.log("JavaScript".indexOf("a")); // 1
console.log("JavaScript".lastIndexOf("a")); // 3

// slice() - extracts section of string
console.log("JavaScript".slice(0, 4)); // "Java"
console.log("JavaScript".slice(4)); // "Script"
console.log("JavaScript".slice(-6)); // "Script"

// split() - splits string into array
console.log("Hello World".split(" ")); // ["Hello", "World"]
console.log("JavaScript".split("")); // ["J","a","v","a","S","c","r","i","p","t"]

// substring() - similar to slice but with some differences
console.log("JavaScript".substring(0, 4)); // "Java"
console.log("JavaScript".substring(4)); // "Script"

// toLowerCase() and toUpperCase() - case conversion
console.log("JavaScript".toLowerCase()); // "javascript"
console.log("JavaScript".toUpperCase()); // "JAVASCRIPT"

// trim() - removes whitespace
console.log("  Hello World  ".trim()); // "Hello World"
console.log("  Hello World  ".trimStart()); // "Hello World  "
console.log("  Hello World  ".trimEnd()); // "  Hello World"

// replace() - replaces substring or pattern
console.log("JavaScript".replace("Java", "Type")); // "TypeScript"
console.log("Hello World".replace(/[aeiou]/g, "")); // "Hll Wrld" (remove vowels)

// match() - finds matches against regex
console.log("Hello 123 World".match(/\d+/)); // ["123"]
console.log("Hello 123 World".match(/\d+/g)); // ["123"]

// padStart() and padEnd() - padding
console.log("5".padStart(3, "0")); // "005"
console.log("Hello".padEnd(10, ".")); // "Hello....."

Characteristics

  • Strings are immutable - methods return new strings
  • Most methods are case-sensitive
  • Many methods accept negative indices (counting from end)
  • Regular expressions can be used with many string methods
  • Template literals (backticks) offer advanced string interpolation

Practical Examples

// Formatting currency
function formatCurrency(amount) {
    return "$" + amount.toFixed(2).replace(/\d(?=(\d{3})+\.)/g, "$&,");
}
console.log(formatCurrency(1234567.89)); // "$1,234,567.89"

// Capitalizing words in a string
function capitalizeWords(str) {
    return str.toLowerCase().replace(/\b\w/g, char => char.toUpperCase());
}
console.log(capitalizeWords("hello world")); // "Hello World"

// Validating email format
function isValidEmail(email) {
    const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
    return emailRegex.test(email);
}
console.log(isValidEmail("test@example.com")); // true
console.log(isValidEmail("invalid.email")); // false

// Extracting digits from a string
function extractDigits(str) {
    return str.match(/\d+/g)?.join("") || "";
}
console.log(extractDigits("Order #1234 for $99.99")); // "12349999"

// Truncating text with ellipsis
function truncateText(text, maxLength) {
    return text.length > maxLength ? text.slice(0, maxLength - 3) + "..." : text;
}
console.log(truncateText("This is a long text that needs truncation", 20)); 
// "This is a long te..."

// Converting string to camelCase
function toCamelCase(str) {
    return str.toLowerCase().replace(/[^a-zA-Z0-9]+(.)/g, (_, chr) => chr.toUpperCase());
}
console.log(toCamelCase("Hello world JavaScript")); // "helloWorldJavaScript"

// Formatting phone number
function formatPhoneNumber(phone) {
    const numbers = phone.replace(/\D/g, "");
    return numbers.replace(/(\d{3})(\d{3})(\d{4})/, "($1) $2-$3");
}
console.log(formatPhoneNumber("1234567890")); // "(123) 456-7890"

// Counting words in a string
function countWords(str) {
    return str.trim().split(/\s+/).filter(word => word.length > 0).length;
}
console.log(countWords("Hello world, this is JavaScript!")); // 5

Common Use Cases

  • Data validation and sanitization
  • User input formatting
  • Text processing and analysis
  • Data extraction from strings
  • Display formatting for numbers and text

Best Practices

  1. Use template literals for string concatenation instead of + operator
  2. Be mindful of case sensitivity when comparing strings
  3. Use parseInt() with radix parameter to avoid unexpected parsing
  4. Consider using Number.isNaN() instead of isNaN() for more precise checking
  5. Use regular expressions for complex string operations
  6. Be cautious with floating point precision in financial calculations
  7. Consider using libraries like Intl for advanced number formatting
